WITH 
MEMBER [Measures].[Faturamento de Vendas]
AS
'[MEASURES].[FATURAMENTO]',
FORMAT_STRING = "#,###.00"

SELECT NON EMPTY
	({
		ORDER(
			[CLIENTE].[HIERARQUIA GEOGRÁFICA].[NÍVEL CIDADE].MEMBERS
			, [Faturamento de Vendas]
			, BDESC)
	})
	ON ROWS,
	({
		[TEMPO].[HIERARQUIA DO MÊS E ANO].[ANO].&[2014]
	}) 
	ON COLUMNS
FROM [COMPLETO]
WHERE [Faturamento de Vendas]
