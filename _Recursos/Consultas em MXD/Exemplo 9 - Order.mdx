SELECT 
	({
		[Produto].[Marca].&[Festival de Sabores],
		[Produto].[Marca].&[Pedaços de Frutas],
		[Produto].[Marca].&[Sabor da Montanha]
	}) 
	ON ROWS ,
	({
		[Tempo].[Ano].&[2014]
	}) ON COLUMNS
FROM [COMPLETO]

SELECT 
	ORDER (
	({
		[Produto].[Marca].&[Festival de Sabores],
		[Produto].[Marca].&[Pedaços de Frutas],
		[Produto].[Marca].&[Sabor da Montanha]
	}) , [Measures].[Faturamento] , DESC)
	
	ON ROWS ,
	({
		[Tempo].[Ano].&[2014]
	}) ON COLUMNS
FROM [COMPLETO]
WHERE [Measures].[Faturamento]
