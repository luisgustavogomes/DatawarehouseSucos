WITH 

MEMBER [MEASURES].[FATURAMENTO DE VENDAS] AS
'[MEASURES].[FATURAMENTO]', FORMAT_STRING = "#,##0.00"

SET [CIDADES] AS
'ORDER([Cliente].[Hierarquia Geográfica].[Nível ], [FATURAMENTO DE VENDAS], BDESC)'

MEMBER [Cliente].[Hierarquia Geográfica].[SOMA] AS 
'SUM([CIDADES])', FORMAT_STRING = "#,##0.00"

MEMBER [MEASURES].[PERCENTUAL] AS
'[MEASURES].[FATURAMENTO DE VENDAS]/([MEASURES].[FATURAMENTO DE VENDAS],[Cliente].[Hierarquia Geográfica].[SOMA])', FORMAT_STRING = "PERCENT"

SELECT 
	NON EMPTY ({
		 [CIDADES]
		 , [Cliente].[Hierarquia Geográfica].[SOMA]
	 }) ON ROWS,
	NON EMPTY ({
		[FATURAMENTO DE VENDAS]
		,[MEASURES].[PERCENTUAL] 
	}) 
	ON COLUMNS
FROM [COMPLETO]
WHERE  [TEMPO].[HIERARQUIA DO MÊS E ANO].[ANO].&[2014]
