WITH SET [Primeiros Meses de 2013] AS

  [Tempo].[Mês e Ano].&[2013]&[01-2013]&[02-2013]&[06-2013]}'

'SUM([Primeiros Meses de 2013],[Measures].[Faturamento Líquido])'
